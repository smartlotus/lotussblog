git fetch origin
git reset --soft origin/main

# 3) 忽略构建产物（防止以后再提交超大文件）
@"
dist/SiteManagerUI/
tools/SiteManagerUI/bin/
tools/SiteManagerUI/obj/
"@ | Add-Content .gitignore

# 4) 从索引移除大文件/构建目录（本地文件不会被删除）
git rm -r --cached --ignore-unmatch dist/SiteManagerUI tools/SiteManagerUI/bin tools/SiteManagerUI/obj

# 5) 重新提交并推送
git add -A
git commit -m "sync site (exclude large build artifacts)"
git push --force-with-lease origin main
